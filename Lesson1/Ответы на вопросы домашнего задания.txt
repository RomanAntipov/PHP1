2. Объяснить, как работает данный код:
<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true?

// оператор сравнения == сравнивает значения переменных $a и $b, не обращая внимания на типы данных. 
// Значение переменной $b (строка '05') преобразуется в целочисленное 5. 
// Что равно $a.


    var_dump((int)'012345');     // Почему 12345?

// Строка '012345' приводится к целочисленному значению.
// Т.е. преобразуется в число 12345, ноль перед ним отбрасывается.

    var_dump((float)123.0 === (int)123.0); // Почему false?

//  (float)123.0 - это число с плавающей точкой 123.0000000...
//  (int)123.0 - приводится к целому числу 123. Т.к. типы данных разные 
// (целочисленный и с плавающей точкой) и проводится сравнение "тождественно равно", 
// в сравниваемые данные не тождественно равны, то результат сравнения - false.

    var_dump((int)0 === (int)'hello, world'); // Почему true?

// сравнение "тождественно равно". В левой части - целочисленный ноль.
// В правой части строковое 'hello, world' преобразуется в целочисленное значение, что даёт ноль.
// поэтому результат сравнения - true

?>
